perf-gate.yml — The Workflow 
name: Performance Gate
on: [push]

jobs:
  audit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Install
        run: npm ci

      - name: 🔦 Run Lighthouse
        run: npx lighthouse $URL \
          --output json \
          --output-path report.json

      - name: 🚦 Performance Gate
        run: node check-score.js
        # exits 1 if score < 80

      - name: 🚀 Deploy
        if: success()
        run: echo "Deploying!"
check-score.js — The Gate 
const fs = require('fs');

// Read the Lighthouse JSON report
const report = JSON.parse(
  fs.readFileSync('report.json', 'utf8')
);

const score =
  report.categories.performance.score * 100;

const THRESHOLD =
  process.env.PERF_THRESHOLD || 80;

console.log(`Score: ${score}/100`);
console.log(`Threshold: ${THRESHOLD}`);

if (score < THRESHOLD) {
  console.error(`❌ GATE FAILED`);
  process.exit(1); // Blocks deploy
}

console.log('✅ Gate passed!');

PERFORMANCE
AS CODE
🚦
Automated Gate
Every push triggers a Lighthouse audit. Scores below threshold automatically block the deploy — no human review needed.
📊
Score History
Every run appends to a scores.json file. Watch your performance trend over time across every commit.
⚙️
Configurable
Set your threshold via the PERF_THRESHOLD env var. Default is 80 — raise it as your team matures.
🔦
Lighthouse Reports
Full HTML Lighthouse reports are saved as pipeline artifacts so you can diagnose exactly what caused a failure.
📝
PR Comments
Score summaries are posted directly to pull requests via $GITHUB_STEP_SUMMARY for instant feedback.
🚀
Zero Overhead
Runs entirely on GitHub Actions free tier. No external services, no API keys, no monthly fees.
